% General macro to draw a shape filled with a bitmap
\newcommand\fillshape[1]{ % #1 = shape
    \begin{scope}
        \clip #1;
    \end{scope}
    \draw[line width=0.5mm] #1;
}

% Particularization for hexagonal shape
\newcommand\fillhexagon{
    \fillshape{(0:4) -- (60:4) -- (120:4) -- (180:4) -- (240:4) -- (300:4) -- cycle}
}

    \node at (0,-2.5) {#4};
    \node at (-2.16506,1.25) {#5};
    \node at (-2.16506,-1.25) {#6};
    
\end{tikzpicture}
}
